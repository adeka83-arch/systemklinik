// ================= SIMPLE DATA ENDPOINTS FOR STATUS CHECKING =================

// Get doctors list for status checking
app.get('/make-server-73417b67/doctors', async (c) => {
  try {
    console.log('👨‍⚕️ Getting doctors list...')
    
    const doctors = await kv.getByPrefix('doctor_')
    
    console.log('✅ Found doctors:', doctors.length)
    return c.json({
      success: true,
      doctors: doctors || [],
      count: doctors?.length || 0
    })
  } catch (error) {
    console.log('❌ Error getting doctors:', error)
    return c.json({ 
      success: false, 
      error: error.message,
      doctors: [],
      count: 0
    }, 500)
  }
})
